144      private int RGBChange(int colour){
145          int output=-300;
146  
147          //while colour isn't a possible amount for RGB
148          while(colour+output<0 || colour+output>255){
149              //randomize number change to the inputted colour
150              output=(rand.nextInt(10)-5)*5;
151          }
152          return  colour+output;
153  
154      }
155      private colour makePaletteColour(colour Colour, String colourName){
156          //make colour and output it
157          colour output=new colour(RGBChange(Colour.red),RGBChange(Colour.green),RGBChange(Colour.blue),colourName);
158          return output;
159      }

161      private colour makePaletteColour(int red, int green,int blue, String colourName){
162          //make colour and output it
