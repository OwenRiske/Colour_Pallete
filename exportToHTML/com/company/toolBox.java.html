101      public static int[] hexCodeToRGB(String hexCode) {
102          String[] splitHexCode = hexCode.split("");
103          int[] output = new int[3];
104          int num = 0;
105          for (int i = 0; i < 3; i++) {
106              double save = 0d;
107  
108              for (int j = 0; j < 2; j++) {
109                  //turn the letters into their number equivalent
110                  if (splitHexCode[i + j].equalsIgnoreCase("a")) {
111                      num = 10;
112                  } else if (splitHexCode[j + i * 2].equalsIgnoreCase("b")) {
113                      num = 11;
114                  } else if (splitHexCode[j + i * 2].equalsIgnoreCase("c")) {
115                      num = 12;
116                  } else if (splitHexCode[j + i * 2].equalsIgnoreCase("d")) {
117                      num = 13;
118                  } else if (splitHexCode[j + i * 2].equalsIgnoreCase("e")) {
119                      num = 14;
120                  } else if (splitHexCode[j + i * 2].equalsIgnoreCase("f")) {
121                      num = 15;
122                  } else {
123                      //the rest turn into numbers
124                      try {
125                          num = Integer.parseInt(splitHexCode[1]);
126                      } catch (Exception e) {
127                          num = 0;
128                      }
129                  }//get numbers into rgb values
130                  save += (8.5 * num);
131              }
132              //turn them into ints
133              output[i] = (int) save;
134          }
135          return output;
136      }

139          //turn hex code into rgb values
